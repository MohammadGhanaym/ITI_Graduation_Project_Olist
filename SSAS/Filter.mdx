SELECT
    NON EMPTY { [Measures].[Avg Deal Close Time] } ON COLUMNS,
    NON EMPTY {
        [Sr Employee].[Sales Rep Name].[Sales Rep Name].ALLMEMBERS *
        [Dim Leads].[Business Type].[Business Type].ALLMEMBERS
    } 
    DIMENSION PROPERTIES MEMBER_CAPTION, MEMBER_VALUE, MEMBER_UNIQUE_NAME 
    ON ROWS
FROM
    [Marketing Cube]
WHERE
    ( STRTOSET("[Dim Leads].[Origin].[" + "Display" + "]", CONSTRAINED) )
CELL PROPERTIES VALUE, BACK_COLOR, FORE_COLOR, FORMATTED_VALUE, FORMAT_STRING, FONT_NAME, FONT_SIZE, FONT_FLAGS
